302  FLOAT64* io_data,
303  USIZE in_data_length,
304  CHAR in_sign
305  )
306 {
307  FLOAT64* data = io_data;
308  USIZE nn = in_data_length;
309  FLOAT64 isign = ( in_sign * 1.0 );
310 
311  USIZE n,mmax,m,j,istep,i;
312  FLOAT64 wtemp,wr,wpr,wpi,wi,theta;
313  FLOAT64 tempr,tempi;
314 
315  n=nn << 1;
316  j=1;
317 
318  for (i=1;i<n;i+=2) {
319  if (j > i) {
320 
321  SWAP(data[j],data[i]);
322  SWAP(data[j+1],data[i+1]);
323  }
324  m=nn;
325  while (m >= 2 && j > m) {
326  j -= m;
327  m >>= 1;
328  }
329  j += m;
330  }
331 
332  mmax=2;
333  while (n > mmax) {
334  istep=mmax << 1;
335  theta=isign*(6.28318530717959/mmax);
336  wtemp=sin(0.5*theta);
337  wpr = -2.0*wtemp*wtemp;
338  wpi=sin(theta);
339  wr=1.0;
340  wi=0.0;
341  for (m=1;m<mmax;m+=2) {
342  for (i=m;i<=n;i+=istep) {
343  j=i+mmax;
344 
345  tempr=wr*data[j]-wi*data[j+1];
346  tempi=wr*data[j+1]+wi*data[j];
347 
348  data[j]=data[i]-tempr;
349  data[j+1]=data[i+1]-tempi;
350 
351  data[i] += tempr;
352  data[i+1] += tempi;
353  }
354  wr=(wtemp=wr)*wpr-wi*wpi+wr;
355  wi=wi*wpr+wtemp*wpi+wi;
356  }
357  mmax=istep;
358  }
