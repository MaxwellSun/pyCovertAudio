186 osx_get_number_of_channels (
187  cahal_device* io_device
188  )
189 {
190  AudioChannelLayout layout_property;
191 
192  UINT32 property_size = sizeof( AudioChannelLayout );
193 
194  io_device->preferred_number_of_channels = 0;
195 
196  OSStatus result =
197  osx_get_device_property_value (
198  io_device->handle,
199  kAudioDevicePropertyPreferredChannelLayout,
200  property_size,
201  &layout_property
202  );
203 
204  if( noErr == result )
205  {
206  CPC_LOG (
207  CPC_LOG_LEVEL_TRACE,
208  "Channel layout tag: 0x%x.",
209  layout_property.mChannelLayoutTag
210  );
211 
212  if( layout_property.mChannelLayoutTag ==
213  kAudioChannelLayoutTag_UseChannelDescriptions )
214  {
215  io_device->preferred_number_of_channels =
216  layout_property.mNumberChannelDescriptions;
217  }
218  else if( layout_property.mChannelLayoutTag !=
219  kAudioChannelLayoutTag_UseChannelBitmap )
220  {
221  // Count the number of bits in the bit map to determine the number of
222  // channels
223 
224  UINT32 channel_bitmap = layout_property.mChannelBitmap;
225 
226  for( UINT32 i = 0; i < sizeof( UINT32 ); i++ )
227  {
228  if( channel_bitmap & 0x0001 )
229  {
230  io_device->preferred_number_of_channels++;
231  }
232 
233  channel_bitmap >>= 1;
234  }
235  }
236  else
237  {
238  io_device->preferred_number_of_channels =
239  AudioChannelLayoutTag_GetNumberOfChannels (
240  layout_property.mChannelLayoutTag
241  );
242  }
243  }
244  else
245  {
246  AudioObjectPropertyAddress property_address =
247  {
248  kAudioDevicePropertyPreferredChannelsForStereo,
249  kAudioObjectPropertyScopeGlobal,
250  kAudioObjectPropertyElementMaster
251  };
252 
253  if( ! AudioObjectHasProperty( io_device->handle, &property_address ) )
254  {
255  io_device->preferred_number_of_channels = 2;
256  }
257  else
258  {
259  result = kAudioHardwareUnknownPropertyError;
260 
261  CPC_LOG (
262  CPC_LOG_LEVEL_WARN,
263  "Property 0x%x is not supported.",
264  kAudioDevicePropertyPreferredChannelsForStereo
265  );
266 
267  CPC_PRINT_CODE( CPC_LOG_LEVEL_WARN, result );
268  }
269  }

271  return( result );
